-- Run this whole file in Supabase SQL Editor.
-- Purpose:
-- 1. Keep old student submissions and choices as history.
-- 2. Allow a student to start a new active application.
-- 3. Count only active applications in HI applicant totals.
-- 4. Keep the existing "Other requirements" company field.
-- 5. Add "Remaining slots" for student-facing availability.

alter table public.companies
add column if not exists other_requirements text not null default '';

alter table public.companies
add column if not exists slots_remaining integer not null default 0;

alter table public.student_submissions
add column if not exists is_active boolean not null default true,
add column if not exists inactive_at timestamptz;

alter table public.student_submissions
drop constraint if exists uq_student_submissions_student_id;

drop index if exists public.uq_student_submissions_one_active;

create unique index uq_student_submissions_one_active
on public.student_submissions (student_id)
where is_active = true;

alter table public.student_choices
drop constraint if exists uq_student_choices_student_company;

drop index if exists public.uq_student_choices_submission_company;

create unique index uq_student_choices_submission_company
on public.student_choices (submission_id, company_id);

drop view if exists public.v_companies_with_slots;

create view public.v_companies_with_slots as
select
  c.id,
  c.company_name,
  c.slots_total,
  c.slots_remaining,
  c.other_requirements,
  c.is_active,
  count(ss.id) as total_applicants
from public.companies c
left join public.student_choices sc
  on sc.company_id = c.id
left join public.student_submissions ss
  on ss.id = sc.submission_id
 and ss.is_active = true
group by c.id, c.company_name, c.slots_total, c.slots_remaining, c.other_requirements, c.is_active;

create or replace function public.start_new_student_application(
    p_student_id bigint,
    p_student_email text
)
returns json
language plpgsql
security definer
set search_path = public
as $$
declare
    v_submission_id bigint;
begin
    if p_student_id is null then
        raise exception 'Student ID is required.';
    end if;

    if p_student_email is null or btrim(p_student_email) = '' then
        raise exception 'Student email is required.';
    end if;

    if not exists (
        select 1
        from public.students s
        where s.id = p_student_id
          and lower(s.email) = lower(btrim(p_student_email))
    ) then
        raise exception 'Student email does not match this account.';
    end if;

    select ss.id
    into v_submission_id
    from public.student_submissions ss
    where ss.student_id = p_student_id
      and ss.is_active = true
    order by ss.submitted_at desc
    limit 1;

    if v_submission_id is null then
        return json_build_object(
            'success', true,
            'started_new_application', false,
            'message', 'No active submitted application found.'
        );
    end if;

    update public.student_submissions
    set
      is_active = false,
      status = 'replaced',
      inactive_at = now()
    where id = v_submission_id;

    return json_build_object(
        'success', true,
        'started_new_application', true,
        'previous_submission_id', v_submission_id,
        'message', 'Previous application kept as history. You can submit a new application.'
    );
end;
$$;

grant execute on function public.start_new_student_application(bigint, text)
to anon, authenticated;

-- Backward-compatible alias for any cached/old portal code that still calls
-- clear_student_preferences. This now archives instead of deleting.
create or replace function public.clear_student_preferences(
    p_student_id bigint,
    p_student_email text
)
returns json
language plpgsql
security definer
set search_path = public
as $$
begin
    return public.start_new_student_application(p_student_id, p_student_email);
end;
$$;

grant execute on function public.clear_student_preferences(bigint, text)
to anon, authenticated;

create or replace function public.submit_student_preferences(
    p_student_id bigint,
    p_company_ids bigint[],
    p_file_url text
)
returns json
language plpgsql
as $$
declare
    v_submission_id bigint;
    v_company_id bigint;
    v_choice_rank integer;
    v_distinct_count integer;
    v_total_count integer;
begin
    if p_student_id is null then
        raise exception 'Student ID is required.';
    end if;

    if p_company_ids is null or array_length(p_company_ids, 1) is null then
        raise exception 'At least one company must be selected.';
    end if;

    if array_length(p_company_ids, 1) > 3 then
        raise exception 'You can only select up to 3 companies.';
    end if;

    if p_file_url is null or btrim(p_file_url) = '' then
        raise exception 'File URL is required.';
    end if;

    select count(distinct x), count(*)
    into v_distinct_count, v_total_count
    from unnest(p_company_ids) as x;

    if v_distinct_count <> v_total_count then
        raise exception 'Duplicate companies are not allowed.';
    end if;

    if exists (
        select 1
        from public.student_submissions
        where student_id = p_student_id
          and is_active = true
    ) then
        raise exception 'You have already submitted your preferences.';
    end if;

    for v_choice_rank in 1..array_length(p_company_ids, 1) loop
        v_company_id := p_company_ids[v_choice_rank];

        if not exists (
            select 1
            from public.companies c
            where c.id = v_company_id
              and c.is_active = true
        ) then
            raise exception 'Selected company does not exist or is inactive. Company ID: %', v_company_id;
        end if;
    end loop;

    insert into public.student_submissions (
        student_id,
        status,
        file_url,
        is_active
    )
    values (
        p_student_id,
        'submitted',
        p_file_url,
        true
    )
    returning id into v_submission_id;

    for v_choice_rank in 1..array_length(p_company_ids, 1) loop
        v_company_id := p_company_ids[v_choice_rank];

        insert into public.student_choices (
            submission_id,
            student_id,
            company_id,
            choice_rank
        )
        values (
            v_submission_id,
            p_student_id,
            v_company_id,
            v_choice_rank
        );
    end loop;

    return json_build_object(
        'success', true,
        'submission_id', v_submission_id,
        'message', 'Preferences submitted successfully.'
    );
end;
$$;

notify pgrst, 'reload schema';
