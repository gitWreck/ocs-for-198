create view public.v_companies_with_slots as
select 
  c.id,
  c.company_name,
  c.slots_total,
  c.slots_remaining,
  c.other_requirements,
  c.is_active,
  count(sc.id) as total_applicants
from public.companies c
left join public.student_choices sc
  on sc.company_id = c.id
group by c.id, c.company_name, c.slots_total, c.slots_remaining, c.other_requirements, c.is_active;

create or replace function public.submit_student_preferences(
    p_student_id bigint,
    p_company_ids bigint[],
    p_file_url text
)
returns json
language plpgsql
as $$
declare
    v_submission_id bigint;
    v_company_id bigint;
    v_choice_rank integer;
    v_distinct_count integer;
    v_total_count integer;
begin
    if p_student_id is null then
        raise exception 'Student ID is required.';
    end if;

    if p_company_ids is null or array_length(p_company_ids, 1) is null then
        raise exception 'At least one company must be selected.';
    end if;

    if array_length(p_company_ids, 1) > 3 then
        raise exception 'You can only select up to 3 companies.';
    end if;

    if p_file_url is null or btrim(p_file_url) = '' then
        raise exception 'File URL is required.';
    end if;

    select count(distinct x), count(*)
    into v_distinct_count, v_total_count
    from unnest(p_company_ids) as x;

    if v_distinct_count <> v_total_count then
        raise exception 'Duplicate companies are not allowed.';
    end if;

    if exists (
        select 1
        from public.student_submissions
        where student_id = p_student_id
    ) then
        raise exception 'You have already submitted your preferences.';
    end if;

    for v_choice_rank in 1..array_length(p_company_ids, 1) loop
        v_company_id := p_company_ids[v_choice_rank];

        if not exists (
            select 1
            from public.companies c
            where c.id = v_company_id
              and c.is_active = true
        ) then
            raise exception 'Selected company does not exist or is inactive. Company ID: %', v_company_id;
        end if;
    end loop;

    insert into public.student_submissions (
        student_id,
        status,
        file_url
    )
    values (
        p_student_id,
        'submitted',
        p_file_url
    )
    returning id into v_submission_id;

    for v_choice_rank in 1..array_length(p_company_ids, 1) loop
        v_company_id := p_company_ids[v_choice_rank];

        insert into public.student_choices (
            submission_id,
            student_id,
            company_id,
            choice_rank
        )
        values (
            v_submission_id,
            p_student_id,
            v_company_id,
            v_choice_rank
        );
    end loop;

    return json_build_object(
        'success', true,
        'submission_id', v_submission_id,
        'message', 'Preferences submitted successfully.'
    );
end;
$$;

create or replace function public.clear_student_preferences(
    p_student_id bigint,
    p_student_email text
)
returns json
language plpgsql
security definer
set search_path = public
as $$
declare
    v_submission_id bigint;
begin
    if p_student_id is null then
        raise exception 'Student ID is required.';
    end if;

    if p_student_email is null or btrim(p_student_email) = '' then
        raise exception 'Student email is required.';
    end if;

    if not exists (
        select 1
        from public.students s
        where s.id = p_student_id
          and lower(s.email) = lower(btrim(p_student_email))
    ) then
        raise exception 'Student email does not match this account.';
    end if;

    select ss.id
    into v_submission_id
    from public.student_submissions ss
    where ss.student_id = p_student_id;

    if v_submission_id is null then
        return json_build_object(
            'success', true,
            'cleared', false,
            'message', 'No submitted preferences found.'
        );
    end if;

    delete from public.student_submissions
    where id = v_submission_id;

    return json_build_object(
        'success', true,
        'cleared', true,
        'message', 'Submitted preferences cleared successfully.'
    );
end;
$$;

grant execute on function public.clear_student_preferences(bigint, text)
to anon, authenticated;

create or replace function public.set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

create trigger trg_student_submissions_updated_at
before update on public.student_submissions
for each row
execute function public.set_updated_at();

create table public.student_submissions (
    id bigint generated always as identity primary key,

    student_id bigint not null references public.students(id) on delete cascade,
    status text not null default 'submitted',
    
    file_url text not null,
    
    submitted_at timestamptz not null default now(),
    updated_at timestamptz not null default now(),

    constraint uq_student_submissions_student_id unique (student_id)
);


----------------------------------


create table public.student_choices (
    id bigint generated always as identity primary key,

    submission_id bigint not null references public.student_submissions(id) on delete cascade,
    student_id bigint not null references public.students(id) on delete cascade,
    company_id bigint not null references public.companies(id) on delete restrict,

    choice_rank integer not null,
    created_at timestamptz not null default now(),

    constraint check_choice_rank_valid
        check (choice_rank in (1, 2, 3)),

    constraint uq_student_choices_submission_rank
        unique (submission_id, choice_rank),

    constraint uq_student_choices_student_company
        unique (student_id, company_id)
);


----------------------------------


create table public.companies (
    id bigint generated always as identity primary key,

    company_name text not null,
    slots_total integer not null default 0,
    slots_remaining integer not null default 0,
    other_requirements text not null default '',
    slots_taken integer not null default 0,

    is_active boolean not null default true,

    created_at timestamptz not null default now()
);


--------------------------


create table public.students (
    id bigint generated always as identity primary key,

    student_no text not null,
    fullname text not null,
    email text not null unique,

    created_at timestamptz not null default now()
);
